class Solution {
    final int GUARD = 4, WALL = 2, EYE = 1;
    public int countUnguarded(int m, int n, int[][] guards, int[][] walls) {
        int[][] map1 = new int[m][n];
        int[][] map2 = new int[m][n];
        for (var guard : guards) {
            map1[guard[0]][guard[1]] = GUARD;
            map2[guard[0]][guard[1]] = GUARD;
        }
        for (var wall : walls) {
            map1[wall[0]][wall[1]] = WALL;
            map2[wall[0]][wall[1]] = WALL;
        }
        for (int[] cur : map1)
            for(int j = 0; j < n;)
                if(cur[j++] == GUARD){
                    int p = j - 2;
                    while(p >= 0 && (cur[p] & 7) == 0)
                        cur[p--] = EYE;
                    while (j < n && (cur[j] & 7) == 0)
                        cur[j++] = EYE;
                }
        for (int i = 0; i < n; i++)
            for(int j = 0; j < m;)
                if(map2[j++][i] == GUARD){
                    int p = j - 2;
                    while(p >= 0 && (map2[p][i] & 7) == 0)
                        map2[p--][i] = EYE;
                    while (j < m && (map2[j][i] & 7) == 0)
                        map2[j++][i] = EYE;
                }
        int ans = 0;
        for(int i = 0; i < m; i++)
            for(int j = 0; j < n; j++)
                if(map2[i][j] + map1[i][j] == 0)
                    ans++;
        return ans;
    }
}

class Solution {
public:
    int countUnguarded(int m, int n, vector<vector<int>>& guards, vector<vector<int>>& walls) {
        vector<vector<char>> grid(m, vector<char>(n));
        for (auto& arr : walls) {
            grid[arr[0]][arr[1]] = 1;
        }
        auto check = [&] (int i, int j) {
            return i >= 0 && j >= 0 && i < m && j < n;
        };
        int move[4][2] {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
        for (auto& arr : guards) {
            for (int k = 0; k < 4; ++k) {
                int ti = arr[0], tj = arr[1];
                while (check(ti, tj) 
                    && (grid[ti][tj] & 1) == 0 
                    && (grid[ti][tj] & (1 << (k + 1))) == 0) {
                    grid[ti][tj] |= (1 << (k + 1));
                    ti += move[k][0];
                    tj += move[k][1];
                }
            }
            grid[arr[0]][arr[1]] |= 1;
        }
        int ans = 0;
        for (auto& arr : grid) {
            for (auto i : arr) {
                if (i == 0) ++ans;
            }
        }
        return ans;
    }
};

class Solution {
    public int countUnguarded(int m, int n, int[][] guards, int[][] walls) {
        int[][] protect = new int[m][n];
        for(int[] rows : protect){
            Arrays.fill(rows, 1);
        }
        for(int i=0; i<walls.length; i++){
            protect[walls[i][0]][walls[i][1]] = 2;
        }
        for(int[] guard : guards){
            int x = guard[0], y = guard[1];
            protect[x][y] = 3;
        }
        
        for(int[] guard : guards){
            int x = guard[0], y = guard[1];
            for(int i = x-1; i >= 0; i--){
                if(protect[i][y] == 2 || protect[i][y] == 3) break;
                protect[i][y] = 0;
            }
            for(int i = x+1; i < m; i++){
                if(protect[i][y] == 2 || protect[i][y] == 3) break;
                protect[i][y] = 0;
            }
            for(int j = y-1; j >= 0; j--){
                if(protect[x][j] == 2 || protect[x][j] == 3) break;
                protect[x][j] = 0;
            }
            for(int j = y+1; j < n; j++){
                if(protect[x][j] == 2 || protect[x][j] == 3) break;
                protect[x][j] = 0;
            }
        }
        
        int ans = 0;
        for(int[] rows : protect){
            for(int num : rows){
                if(num==1){
                    ans+=1;
                }
            }
        }
        return ans;
    }
}

var countUnguarded = function (m, n, guards, walls) {
    // å¢™-1ï¼Œç»¿è‰²0
    const grid = Array.from({ length: m }, () => Array(n).fill(0));
    for (const [x, y] of walls) {
        grid[x][y] = -1;
    }
    const DIRS = [
        [0, -1],
        [0, 1],
        [-1, 0],
        [1, 0],
    ];
    const DMASK = [1 | 2, 1 | 2, 4 | 8, 4 | 8];
    for (const [x, y] of guards) {
        grid[x][y] = 15; // 1 | 2 | 4 | 8
        for (let idx = 0; idx < DIRS.length; idx++) {
            const dmask = DMASK[idx];
            const dx = DIRS[idx][0];
            const dy = DIRS[idx][1];
            let i = x + dx;
            let j = y + dy;
            while (i >= 0 && i < m && j >= 0 && j < n) {
                // å¢™æˆ–è€…è¯¥æ ¼å­è¢«ç›¸é‚»å®ˆå«æ ‡è®°
                if (grid[i][j] === -1 || grid[i][j] & dmask) break;
                grid[i][j] |= dmask;
                i += dx;
                j += dy;
            }
        }
    }

    let ans = 0;
    for (let i = 0; i < m; i++) {
        for (let j = 0; j < n; j++) {
            if (grid[i][j] === 0) ans++;
        }
    }

    return ans;
};

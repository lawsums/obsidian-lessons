nums = []

def dfs(n: int, cur: int, queue: List[int], pre: int) -> int:
    if cur == 0 and n == 0:
        nums.append(pre)
        return
    for num, cnt in enumerate(queue[1:], 1):
        if cnt == 0: continue
        if num > cnt:
            queue[num] -= 1
            dfs(n, cur - 1, queue, pre * 10 + num)
            queue[num] += 1
        else:
            if num > n: continue
            queue[num] -= 1
            dfs(n - num, cur - 1, queue, pre * 10 + num)
            queue[num] += 1

dfs(1, 1, [0,1], 0)
dfs(2, 2, [0,1,2], 0)

    def nextBeautifulNumber(self, n: int) -> int:

        return nums[i]
